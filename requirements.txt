# SETUP INSTRUCTIONS (order matters due to JAX / PyTorch coexistence):
#
#   1. Create conda environment:
#        conda create -n libero python=3.11 -y
#        conda activate libero
#
#   2. Install C compiler (needed by Triton / vLLM CUDA graph compilation):
#        conda install -c conda-forge gcc_linux-64 gxx_linux-64 -y
#        # Then create symlinks in $CONDA_PREFIX/bin:
#        ln -sf $CONDA_PREFIX/bin/x86_64-conda-linux-gnu-gcc $CONDA_PREFIX/bin/gcc
#        ln -sf $CONDA_PREFIX/bin/x86_64-conda-linux-gnu-g++ $CONDA_PREFIX/bin/g++
#        ln -sf $CONDA_PREFIX/bin/gcc $CONDA_PREFIX/bin/cc
#
#   3. Install OpenGL libs (for MuJoCo headless rendering):
#        conda install -c conda-forge libopengl mesalib -y
#
#   4. Install PyTorch (CUDA 12.8):
#        pip install torch==2.9.0 torchvision==0.24.0 torchaudio==2.9.0 \
#            --index-url https://download.pytorch.org/whl/cu128
#
#   5. Install pinned JAX + numpy (must come before openpi):
#        pip install "jax[cuda12]==0.5.3" "numpy>=1.26,<2.0"
#
#   6. Install this requirements file:
#        pip install -r requirements.txt
#
#   7. Install local editable packages (adjust paths to your machine):
#        pip install -e /path/to/LIBERO    --no-deps
#        pip install -e /path/to/openpi    --no-deps
#        pip install    openpi-client      --no-deps
#
#   8. (Optional) Set env vars in your .env or shell:
#        CC=$CONDA_PREFIX/bin/gcc
#        XLA_PYTHON_CLIENT_MEM_FRACTION=0.25
#
# ============================================================================
